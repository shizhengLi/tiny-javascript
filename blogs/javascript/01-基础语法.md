# JavaScript 基础语法介绍

JavaScript 是一种动态类型的编程语言，主要用于网页开发。本文将介绍 JavaScript 的基础语法，帮助你快速入门。

## 变量声明

JavaScript 有三种声明变量的方式：

```javascript
// var (不推荐，作用域问题)
var message = "Hello World";

// let (推荐，块级作用域)
let name = "张三";
let age = 25;

// const (常量，不可重新赋值)
const PI = 3.14159;
const BIRTH_YEAR = 1998;
```

## 数据类型

### 基本数据类型

```javascript
// 1. String (字符串)
let text = "Hello";
let text2 = 'World';
let template = `Hello, ${name}!`; // 模板字符串

// 2. Number (数字)
let integer = 42;
let float = 3.14;
let negative = -100;

// 3. Boolean (布尔值)
let isActive = true;
let isComplete = false;

// 4. Undefined (未定义)
let notDefined;
console.log(notDefined); // undefined

// 5. Null (空值)
let emptyValue = null;

// 6. Symbol (符号，ES6新增)
let id = Symbol('id');

// 7. BigInt (大整数，ES2020新增)
let bigNumber = 9007199254740991n;
```

### 复杂数据类型

```javascript
// 1. Object (对象)
let person = {
    name: "李四",
    age: 30,
    city: "北京"
};

// 2. Array (数组)
let numbers = [1, 2, 3, 4, 5];
let fruits = ["苹果", "香蕉", "橙子"];

// 3. Function (函数)
function greet(name) {
    return `你好，${name}！`;
}
```

## 运算符

### 算术运算符

```javascript
let a = 10;
let b = 3;

console.log(a + b);  // 13 (加法)
console.log(a - b);  // 7  (减法)
console.log(a * b);  // 30 (乘法)
console.log(a / b);  // 3.333... (除法)
console.log(a % b);  // 1  (取余)
console.log(a ** b); // 1000 (幂运算)

// 增量/减量运算符
let count = 0;
count++;  // count = 1
count--;  // count = 0
++count;  // count = 1
--count;  // count = 0
```

### 比较运算符

```javascript
let x = 5;
let y = '5';

console.log(x == y);   // true (值相等)
console.log(x === y);  // false (值和类型都相等)
console.log(x != y);   // false (值不等)
console.log(x !== y);  // true (值或类型不等)
console.log(x > 3);    // true
console.log(x < 10);   // true
console.log(x >= 5);   // true
console.log(x <= 5);   // true
```

### 逻辑运算符

```javascript
let p = true;
let q = false;

console.log(p && q);  // false (逻辑与)
console.log(p || q);  // true (逻辑或)
console.log(!p);      // false (逻辑非)
```

## 控制结构

### 条件语句

```javascript
// if-else 语句
let score = 85;

if (score >= 90) {
    console.log("优秀");
} else if (score >= 80) {
    console.log("良好");
} else if (score >= 60) {
    console.log("及格");
} else {
    console.log("不及格");
}

// switch 语句
let day = 3;
switch (day) {
    case 1:
        console.log("星期一");
        break;
    case 2:
        console.log("星期二");
        break;
    case 3:
        console.log("星期三");
        break;
    default:
        console.log("其他");
}

// 三元运算符
let message = score >= 60 ? "及格" : "不及格";
```

### 循环语句

```javascript
// for 循环
for (let i = 0; i < 5; i++) {
    console.log(`第 ${i + 1} 次循环`);
}

// while 循环
let count = 0;
while (count < 3) {
    console.log(`count: ${count}`);
    count++;
}

// do-while 循环
let num = 1;
do {
    console.log(`数字: ${num}`);
    num++;
} while (num <= 3);

// for...of 循环 (遍历数组)
let colors = ["红色", "绿色", "蓝色"];
for (let color of colors) {
    console.log(color);
}

// for...in 循环 (遍历对象)
let car = {
    brand: "Toyota",
    model: "Camry",
    year: 2020
};

for (let key in car) {
    console.log(`${key}: ${car[key]}`);
}
```

## 函数

### 函数声明

```javascript
// 函数声明
function add(a, b) {
    return a + b;
}

// 函数表达式
const multiply = function(a, b) {
    return a * b;
};

// 箭头函数 (ES6)
const divide = (a, b) => a / b;

// 带默认参数的函数
function greet(name = "访客") {
    return `你好，${name}！`;
}

// 剩余参数
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}
```

## 对象操作

```javascript
// 对象创建和访问
let student = {
    name: "王五",
    age: 20,
    grades: [85, 90, 78, 92]
};

// 访问属性
console.log(student.name);
console.log(student['age']);

// 添加属性
student.major = "计算机科学";

// 删除属性
delete student.age;

// 对象方法
let calculator = {
    add: function(a, b) {
        return a + b;
    },
    subtract: (a, b) => a - b
};
```

## 数组操作

```javascript
// 数组创建
let arr1 = [1, 2, 3];
let arr2 = new Array(4, 5, 6);

// 数组方法
let fruits = ["苹果", "香蕉", "橙子"];

// 添加元素
fruits.push("葡萄");  // 末尾添加
fruits.unshift("西瓜"); // 开头添加

// 删除元素
fruits.pop();      // 删除末尾元素
fruits.shift();    // 删除开头元素

// 数组遍历
fruits.forEach((fruit, index) => {
    console.log(`${index}: ${fruit}`);
});

// 数组映射
let numbers = [1, 2, 3, 4];
let doubled = numbers.map(num => num * 2);

// 数组过滤
let evenNumbers = numbers.filter(num => num % 2 === 0);

// 数组查找
let found = numbers.find(num => num > 2);
```

## 字符串操作

```javascript
let text = "Hello, JavaScript!";

// 字符串长度
console.log(text.length);

// 字符串方法
console.log(text.toUpperCase());        // 转大写
console.log(text.toLowerCase());        // 转小写
console.log(text.substring(0, 5));     // 提取子串
console.log(text.split(', '));         // 分割字符串
console.log(text.indexOf('JavaScript')); // 查找子串位置

// 字符串模板
let name = "Alice";
let age = 25;
let info = `姓名: ${name}, 年龄: ${age}`;
```

## 类型转换

```javascript
// 字符串转数字
let strNum = "123";
let num1 = Number(strNum);
let num2 = parseInt(strNum);
let num3 = parseFloat("3.14");

// 数字转字符串
let numToStr = String(123);
let numToStr2 = 456.toString();

// 布尔值转换
console.log(Boolean(1));        // true
console.log(Boolean(0));        // false
console.log(Boolean(""));       // false
console.log(Boolean("hello"));  // true
```

## 错误处理

```javascript
// try-catch 语句
try {
    let result = riskyOperation();
    console.log(result);
} catch (error) {
    console.error("发生错误:", error.message);
} finally {
    console.log("无论如何都会执行");
}

// 抛出错误
function validateAge(age) {
    if (age < 0) {
        throw new Error("年龄不能为负数");
    }
    if (age > 150) {
        throw new Error("年龄不能超过150岁");
    }
    return true;
}
```
